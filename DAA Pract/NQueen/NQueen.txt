#include<stdio.h>
#include<conio.h>
#include<math.h>

void Queen(int,int);
int place(int,int);
int board[20],i;

int main()
{
    int n;
    clrscr();

    printf("\nEnter How Many Queen You Want To Place  :-  ");
    scanf("%d",&n);

    Queen(1,n);

    getch();
    return 0;
}
void Queen(int row ,int n)
{
    int column;

    for(column=1;column<=n;column++)
    {
        if(place(row,column))
        {
            board[row]=column;
            if(row==n)
            {
                printf("\nSolution  :-  ");
                for(i=1;i<=n;i++)
                {
                    printf("\t%d",board[i]);
                }
            }
            else
            {
                Queen(row+1,n);
            }
        }
    }
}

int place(int row,int column)
{
    for(i=1;i<row;i++)
    {
        if(board[i]==column)
        {
            return 0;
        }
        if(abs(board[i]-column)==abs(i-row))
        {
            return 0;
        }
    }
    return 1;
}
